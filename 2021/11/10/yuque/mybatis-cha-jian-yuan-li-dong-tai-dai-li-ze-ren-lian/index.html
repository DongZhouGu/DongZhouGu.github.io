public class TargetProxy implements InvocationHandler {
    private Object target;
    private MyInterceptor myInterceptor;

    public TargetProxy(Object target,MyInterceptor myInterceptor) {
        this.target = target;
        this.myInterceptor = myInterceptor;
    }

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        Invocation invocation = new Invocation(target, method, args);
        return myInterceptor.intercept(invocation);
    }

    public static <T> T wrap(T target, MyInterceptor interceptor) {
        TargetProxy targetProxy = new TargetProxy(target, interceptor);
        return (T) Proxy.newProxyInstance(target.getClass().getClassLoader(), target.getClass().getInterfaces(),targetProxy);
    }
}
